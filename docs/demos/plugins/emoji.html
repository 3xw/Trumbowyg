The code

    
// This function initialize emojify.js
function initEmoji() {
    emojify.setConfig({
        img_dir : '../../../bower_components/emojify.js/dist/images/basic/',
    });
    emojify.run();  
}
initEmoji();

$('#editor')
.trumbowyg({
    btns: [
        ['emoji']
    ]
});

// You can also add an emoticon directly on the editor
$('.trumbowyg-editor').on('input propertychange', function() {
    initEmoji();
});
