Noembed plugin for Trumbowyg

You can insert embeded content via a noembed proxy. Plugin uses noembed by default.

The code

    
$('#editor-dropdown')
.trumbowyg({
    btnsDef: {
        // Create a new dropdown
        link: {
            dropdown: ['insertLink', 'removeLink', 'noembed'],
            ico: 'insertLink'
        }
    },
    // Redefine the button pane
    btns: ['viewHTML',
            '|', 'formatting',
            '|', 'btnGrp-semantic',
            '|', 'link',
            '|', 'insertImage',
            '|', 'btnGrp-justify',
            '|', 'btnGrp-lists',
            '|', 'horizontalRule']
});
